# Checkbox (Reactive Form Control)

## Description

A part of AdaptRx- form controls. Checkboxes allow users to select one or more options from a set. They support checked, unchecked, and indeterminate states.

## Import

```typescript
import {AdaptRxCheckboxModule} from '@bmc-ux/adapt-angular';
```

## Component Name / Selector

**Selector:** `adapt-rx-checkbox`

## Properties

### @Inputs

| Name | Type | Default | Description |
|------|------|---------|-------------|
| role | string | 'checkbox' | Native input [role] (Since 14.21.0) |
| value | string | - | Native input [value] |
| checked | boolean | - | Toggle [checked] attribute |
| indeterminate | boolean | - | Toggle [indeterminate] attribute |
| disabledStyleForReadonlyState | boolean | - | Use disabled style for the control if [readonly]=true |
| tooltip | AdaptRxControlLabelTooltip | - | Label icon with a tooltip or popover |
| warningMessage | string | - | Warning message |
| name | string | - | Native control name |
| label | string | - | Control label text |
| subLabel | string | - | Control subLabel text |
| requiredLabel | string | - | Required label text |
| ariaLabel | string | - | Control [aria-label] attribute text |
| ariaLabelledby | string | - | Control [aria-labeledby] attribute text |
| ariaDescribedBy | string | - | Control [aria-describedby] attribute text |
| tabIndex | number | - | Control [tabindex] value |
| testID | string | - | String for test id data attribute |
| id | string | - | Control [id]. Returns autogenerated if the value wasn't set |
| disabled | boolean | - | Control [disabled] attribute |
| readonly | boolean | - | Control [readonly] attribute |
| autofocus | boolean | - | Control [autofocus] attribute |

### @Outputs

| Name | Type | Description |
|------|------|-------------|
| onFocus | EventEmitter<FocusEvent> | Focus emitter |
| onBlur | EventEmitter<FocusEvent> | Blur emitter |
| change | EventEmitter<boolean> | Emits when checkbox value changes |

## Simple Example

### TypeScript Component

```typescript
import {Component} from '@angular/core';
import {FormControl, FormGroup} from '@angular/forms';

@Component({
  selector: 'app-checkbox-demo',
  templateUrl: './checkbox-demo.html'
})
export class CheckboxDemoComponent {
  // Template-driven
  acceptTerms: boolean = false;
  isEnabled: boolean = true;
  indeterminateState: boolean = false;

  // Reactive forms
  form = new FormGroup({
    newsletter: new FormControl(false),
    notifications: new FormControl(true)
  });

  onCheckboxChange(value: boolean): void {
    console.log('Checkbox changed:', value);
  }
}
```

### HTML Template

```html
<!-- Template-driven Form -->
<adapt-rx-checkbox [(ngModel)]="acceptTerms"
                   [label]="'Accept terms and conditions'"
                   (change)="onCheckboxChange($event)"></adapt-rx-checkbox>

<adapt-rx-checkbox [(ngModel)]="isEnabled"
                   [label]="'Enable feature'"
                   [subLabel]="'Additional information about this option'"></adapt-rx-checkbox>

<!-- Indeterminate State -->
<adapt-rx-checkbox [(ngModel)]="isEnabled"
                   [indeterminate]="indeterminateState"
                   [label]="'Select All'"></adapt-rx-checkbox>

<!-- Reactive Form -->
<form [formGroup]="form">
  <adapt-rx-checkbox formControlName="newsletter"
                     [label]="'Subscribe to newsletter'"></adapt-rx-checkbox>
  
  <adapt-rx-checkbox formControlName="notifications"
                     [label]="'Enable notifications'"
                     [disabled]="false"></adapt-rx-checkbox>
</form>

<!-- With Tooltip -->
<adapt-rx-checkbox [(ngModel)]="acceptTerms"
                   [label]="'Privacy policy'"
                   [tooltip]="{text: 'Click to read our privacy policy'}"></adapt-rx-checkbox>

<!-- Readonly State -->
<adapt-rx-checkbox [ngModel]="true"
                   [readonly]="true"
                   [label]="'Read-only checkbox'"></adapt-rx-checkbox>
```

## Key Features

- Three states: checked, unchecked, and indeterminate
- Template-driven and reactive forms support
- Label and sub-label support
- Tooltip/popover integration
- Warning messages
- Readonly and disabled states
- Accessibility support (ARIA attributes)
- Custom styling for readonly state
- Focus/blur events
- Test ID support
