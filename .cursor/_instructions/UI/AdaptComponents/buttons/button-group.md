# Button Group

## Description

Button Group is a component that groups multiple buttons together with shared behavior and styling.

### Notes
- Can be configured to behave as either radio-buttons or checkboxes.
- Currently, there are three types of button-group available - 'button', 'color-picker' and 'tile'.

## Import

```typescript
import {AdaptButtonGroupModule} from '@bmc-ux/adapt-angular';
```

## Component Name / Selector

**Selector:** `adapt-button-group`

**Exported as:** `adaptBtnGroup`

## Properties

### @Inputs

| Name | Type | Default | Description |
|------|------|---------|-------------|
| config | ButtonConfig[] | [] | Configuration for button-group elements. See interface below. |
| size | ButtonSize | - | Specifies buttons size. Supported values: 'xtra-large', 'xtra-small', 'small', 'large', 'block'. Only valid for the button group of type 'button' |
| groupType | ButtonGroupType | 'button' | Specifies type of the button group. Supported values: 'button', 'tile', 'color-picker'. (Since v10.18.0) |
| isVertical | boolean | false | Makes a set of buttons appear vertically stacked. Only valid for the button group of type 'button' (Since v10.11.0) |
| multiselectable | boolean | false | Set multiselectable (checkbox) behaviour. |
| uncheckable | boolean | false | Whether button could be unchecked. |
| useCheckboxStyle | boolean | - | Set checkbox style for multiselectable false. Only valid for the button group of type 'color-picker' and 'tile' (Since v11.14.0) |
| tooltip | AdaptRxControlLabelTooltip | - | Label icon with a tooltip or popover |
| warningMessage | string | - | Warning message |
| name | string | - | Native control name |
| label | string | - | Control label text |
| subLabel | string | - | Control subLabel text |
| requiredLabel | string | - | Required label text |
| ariaLabel | string | - | Control [aria-label] attribute text |
| ariaLabelledby | string | - | Control [aria-labeledby] attribute text |
| ariaDescribedBy | string | - | Control [aria-describedby] attribute text |
| tabIndex | number | - | Control [tabindex] value |
| testID | string | - | String for test id data attribute. For test purposes |
| id | string | - | Control [id]. Returns autogenerated if the value wasn't set |
| disabled | boolean | - | Control [disabled] attribute. No needed if the control uses as reactive one |
| readonly | boolean | - | Control [readonly] attribute |
| autofocus | boolean | - | Control [autofocus] attribute |
| ariaErrorMessage | string | - | Control [aria-errormessage] attribute value. Important: 'aria-errormessage' attribute works in tandem with 'aria-invalid'. Additional info: https://www.w3.org/TR/wai-aria/#aria-errormessage (Since 11.3.0) |

### @Outputs

| Name | Type | Description |
|------|------|-------------|
| modelArrayChanged | EventEmitter<InnerConfigType[]> | Emits an event when button-group state changed |
| onFocus | EventEmitter<FocusEvent> | Focus emitter |
| onBlur | EventEmitter<FocusEvent> | Blur emitter |

|------|------|-------------|
| name | string | Button title |
| value | string \| number | - |

| context | KeyValueObject | A context object which can be passed to customize the swatch element template (Since v10.18.0) |
| background | string | Button background color. Only valid for the button group of type 'tile' and 'color-picker' (Since v10.18.0) |
| customClass | string | Custom css class to be added to the button. Only valid for the button group of type 'tile' and 'color-picker' (Since v10.18.0) |

## Simple Example

### TypeScript Component

```typescript
import {Component, OnInit} from '@angular/core';
import {ButtonConfig, ButtonGroupType, ButtonSize} from '@bmc-ux/adapt-angular';

@Component({
  selector: 'adapt-button-group-demo',
  templateUrl: './button-group.demo.html'
})
export class AdaptButtonGroupDemoComponent implements OnInit {
  verticalBtnGroup: boolean = false;
  disabledBtnGroup: boolean = false;
  multiselectableBtnGroup: boolean = true;
  uncheckableBtnGroup: boolean = true;
  
  currentButtonSize: ButtonSize | 'default' = 'default';
  currentGroupType: ButtonGroupType = 'button';
  
  model: boolean[] = [true, false, true, false, false];
  groupConfig: ButtonConfig[];

  ngOnInit(): void {
    this.groupConfig = [
      {
        name: 'Primary 1',
        value: 'left',
      },
      {
        name: 'Primary 2',
        value: 'center',
      },
      {
        name: 'Primary 3',
        value: 'center-center',
      },
      {
        name: 'Primary 4',
        value: 'right',
      },
      {
        name: 'Primary 5',
        value: 'right-right',
      }
    ];
  }
  
  arrayCallback(event: any): void {
    console.log('Button group state changed:', event);
  }
}
```

### HTML Template

```html
<adapt-button-group [(ngModel)]="model"
                    [config]="groupConfig"
                    [groupType]="currentGroupType"
                    [disabled]="disabledBtnGroup"
                    [multiselectable]="multiselectableBtnGroup"
                    [uncheckable]="uncheckableBtnGroup"
                    [size]="currentButtonSize"
                    [ariaLabelledby]="'group-label'"
                    [isVertical]="verticalBtnGroup"
                    (modelArrayChanged)="arrayCallback($event)"></adapt-button-group>
<span class="sr-only" id="group-label">Button group with {{currentGroupType}} type</span>
```

### Button Group Types

- **button:** Standard button group with text labels
- **color-picker:** Visual color selection buttons with background colors
- **tile:** Custom tile buttons using templates for complex content
